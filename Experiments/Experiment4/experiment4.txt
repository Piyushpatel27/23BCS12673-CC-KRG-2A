class Solution {
  public:
    int sumBitDifferences(vector<int>& arr) {
        int ans = 0, n = arr.size();

        for (int i = 0; i < 32; i++) {
            int count = 0;
            for (int j = 0; j < n; j++){
                if ((arr[j] & (1 << i)))    count++;
            }
            ans += (count * (n - count) * 2);
        }
        return ans;
    }
};
