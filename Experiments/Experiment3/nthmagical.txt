class Solution {
public:
    long long gcd(long long a,long long b){
        if(b==0) return a;
        return gcd(b,a%b);
    }

    long long lcm(long long a,long long b){
        return a*b/gcd(a,b);
    }

    int nthMagicalNumber(int n, int a, int b) {
        long long low = 1;
        long long high = 1e18;
        long long mod = 1e9 + 7;
        
        while(low<high){
            long long mid = low+(high-low)/2;
            long long count = mid/a+ mid/b -mid/lcm(a,b);
            if(count<n){
                low = mid+1;
            } 
            else{
                high = mid;
            }
        }
        return low%mod;
    }
};
